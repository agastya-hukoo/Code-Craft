using System.Windows.Forms;

public class MainForm : Form
{
    public MainForm()
    {
        // Initialize your form's components
        InitializeComponent();
        
        // Apply the custom font to all controls on this form
        ApplyCustomFont(this);
    }

    private void ApplyCustomFont(Control control)
    {
        // Apply the font to the control if it's a type that contains text
        if (control is Button || control is Label) // Add more control types as needed
        {
            control.Font = CustomFontProvider.GetCustomFont(12); // Specify your desired size
        }

        // Recursively apply the font to child controls
        foreach (Control child in control.Controls)
        {
            ApplyCustomFont(child);
        }
    }

    // Your form's other methods and events
}
